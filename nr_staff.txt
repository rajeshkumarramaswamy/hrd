CREATE TABLE nr_staff (
  id int(10) unsigned NOT NULL auto_increment,
  employee_number smallint(5) default NULL,
  employee_id varchar(10) default NULL,
  user_name varchar(150) default NULL,
  pwd varchar(100) default NULL,
  staff_fname varchar(25) NOT NULL default '',
  staff_lname varchar(25) NOT NULL default '',
  staff_email varchar(82) NOT NULL default '',
  staff_alternate_email varchar(82) default NULL,
  dept_id int(2) unsigned NOT NULL default '0',
  roster_id int(2) default NULL,
  grp_id int(2) unsigned NOT NULL default '0',
  desig_id int(2) default '0',
  swstaff_id int(10) default NULL,
  staff_resource_type tinyint(1) unsigned NOT NULL default '0',
  is_admin tinyint(1) NOT NULL default '0',
  is_ad_admin tinyint(4) default '0',
  nocid int(2) unsigned NOT NULL,
  manupulate_staff_roster int(2) unsigned default '0',
  alltickets_acess int(2) unsigned default '0',
  is_active smallint(5) default '1',
  is_found smallint(5) default '0',
  reporting_manager_id int(10) default NULL,
  date_of_join date default NULL,
  phone_number varchar(15) default NULL,
  skype_id varchar(50) default NULL,
  alternate_number varchar(15) default NULL,
  phone_priority tinyint(3) default '1',
  escalation_adm tinyint(4) default '0',
  is_ic_admin tinyint(1) default '0',
  is_tb_admin tinyint(1) default '0',
  total_exp float(4,2) default NULL,
  is_mngr_dboard_access tinyint(4) default '0',
  sdmtracker_access tinyint(4) default '0',
  hrdata_access tinyint(4) default '0',
  PRIMARY KEY  (id),
  UNIQUE KEY staff_email (staff_email),
  UNIQUE KEY user_name (user_name),
  KEY FK_nr_staff_1 (dept_id),
  KEY FK_nr_staff_2 (grp_id),
  KEY FK_nr_staff_3 (desig_id),
  CONSTRAINT FK_nr_staff_1 FOREIGN KEY (dept_id) REFERENCES nr_departments (id),
  CONSTRAINT FK_nr_staff_2 FOREIGN KEY (grp_id) REFERENCES nr_groups (id));
